function toggleProfileOptions() { const rabbitType = document.getElementById('rabbitType').value; // Hide all sections initially document.getElementById('fattenerOptions').style.display = 'none'; document.getElementById('breederOptions').style.display = 'none'; document.getElementById('doeOptions').style.display = 'none'; document.getElementById('pregnancyDetailsDoe').style.display = 'none'; // Show fields based on selected profile if (rabbitType === 'fatteners') { document.getElementById('fattenerOptions').style.display = 'block'; } else if (rabbitType === 'breeder') { document.getElementById('breederOptions').style.display = 'block'; } else if (rabbitType === 'doe') { document.getElementById('doeOptions').style.display = 'block'; } } function togglePregnancyDetailsDoe(isPregnant) { document.getElementById('pregnancyDetailsDoe').style.display = isPregnant ? 'block' : 'none'; } // Add an event listener for pregnancy status change document.querySelectorAll('input[name="pregnancyStatus"]').forEach((input) => { input.addEventListener('change', function () { togglePregnancyDetailsDoe(this.value === 'pregnant'); }); });
